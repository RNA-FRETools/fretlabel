[6]:

mod = lmfit.models.GaussianModel()
pars = mod.guess(hist, x=bincenters)
out = mod.fit(hist, pars, x=bincenters)
x_fit = np.linspace(0,1,100)
y_fit = mod.func(x_fit, *list(out.params.valuesdict().values())[:3])
